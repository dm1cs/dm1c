&НаСервере
Процедура ЗагрузитьКлиентовНаСервере(Результат, ДопПараметр) Экспорт
	
	Если НЕ(Результат = Неопределено) Тогда    
	    СсылкаНаФайл = Результат[0];
	Иначе                                
	    Сообщить("Файл не выбран!");
	    Возврат;
	КонецЕсли; 
	
	ТабДок = Новый ТабличныйДокумент;
	ТабДок.Прочитать(СсылкаНаФайл);
	
	Для Строка = 2  По ТабДок.ВысотаТаблицы Цикл
		ДанныеФИО = Справочники.Клиенты.СоздатьЭлемент();		
		ДанныеФИО.Наименование = ТабДок.Область(Строка,2).Текст;
			
	   ДанныеФИО.Записать();
	   Сообщить("Информация добавлена" + ДанныеФИО.Наименование); 
   КонецЦикла;
